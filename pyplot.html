import matplotlib.pyplot as plt
import numpy as np

# Generate a sequence of at least 5 random numbers
random_numbers = np.random.randint(0, 100, size=5)

# Generate x values for the plot
x_values = np.arange(len(random_numbers))

# Create a figure and subplots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 6))

# Scatter plot
ax1.scatter(x_values, random_numbers, color='blue', label='Random Numbers')
ax1.set_title('Scatter Plot')
ax1.set_xlabel('Index')
ax1.set_ylabel('Value')
ax1.legend()

# Line plot
ax2.plot(x_values, random_numbers, color='red', marker='o', linestyle='-', label='Random Numbers')
ax2.set_title('Line Plot')
ax2.set_xlabel('Index')
ax2.set_ylabel('Value')
ax2.legend()

plt.tight_layout()
plt.show()
